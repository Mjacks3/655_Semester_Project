# Coefficient of Detemination (returned from .score) : 
# It is interpreted as the proportion of the variance in the dependent variable 
#that is predictable from the independent variable.

#   x <0 : extrememly poor fit
#   x==0  :  means that the dependent variable cannot be predicted from the independent variable.
#  0<x<1 : between 0 and 1 indicates the extent to which the dependent variable is predictable.
            #An R2 of 0.10 means that 10 percent of the variance in Y is predictable from X; 
#   x==1 :   means the dependent variable can be predicted without error from the independent variable.

#  Iterate N times: Randomly Choose train/test set, create models, fit and train, get R^2 and MSE values. 
# (get coefficients where easily accessible [Just for additional analysis] ), then after loop, average R^2, MSE values


N = 1000
results_dict = {}
results_dict["Linear Regression"] = [0.0,0.0,0]
results_dict["Lasso Regression"] = [0.0,0.0,0]
results_dict["Ridge Regression"] = [0.0,0.0,0]
results_dict["Polynimal Deg 2 Regression"] = [0.0,0.0]
results_dict["Polynimal Deg 3 Regression"] = [0.0,0.0]
results_dict["Polynimal Deg 4 Regression"] = [0.0,0.0]
results_dict["Polynimal Deg 5 Regression"] = [0.0,0.0]


for newset in range(0,N):
    full_train_df = df
    test_df =  pd.DataFrame({})
    ran_samples =  random.sample(range(1,12), 3)

    for rands in ran_samples:
        test_df = test_df.append(df.iloc[[rands]])
        full_train_df = full_train_df.drop(df.index[[rands]])

    
    lin_reg = LinearRegression()
    las_reg = Lasso()
    ridge_reg = Ridge()
    two_reg = svm.SVR(kernel='poly', degree=2)
    three_reg = svm.SVR(kernel='poly', degree=3)
    four_reg = svm.SVR(kernel='poly', degree=4)
    five_reg = svm.SVR(kernel='poly', degree=5)

    
    
    lin_reg.fit(full_train_df.drop(['Net Income'], axis=1).values, full_train_df['Net Income'].values)
    las_reg.fit(full_train_df.drop(['Net Income'], axis=1).values, full_train_df['Net Income'].values)
    ridge_reg.fit(full_train_df.drop(['Net Income'], axis=1).values, full_train_df['Net Income'].values)
    two_reg.fit(full_train_df.drop(['Net Income'], axis=1).values, full_train_df['Net Income'].values)
    three_reg.fit(full_train_df.drop(['Net Income'], axis=1).values, full_train_df['Net Income'].values)
    four_reg.fit(full_train_df.drop(['Net Income'], axis=1).values, full_train_df['Net Income'].values)
    five_reg.fit(full_train_df.drop(['Net Income'], axis=1).values, full_train_df['Net Income'].values)
    
    results_dict["Linear Regression"][0] +=  lin_reg.score(test_df.drop(['Net Income'], axis=1).values, test_df['Net Income'].values)
    results_dict["Lasso Regression"][0] +=  las_reg.score(test_df.drop(['Net Income'], axis=1).values, test_df['Net Income'].values)
    results_dict["Ridge Regression"][0] +=  ridge_reg.score(test_df.drop(['Net Income'], axis=1).values, test_df['Net Income'].values)
    results_dict["Polynimal Deg 2 Regression"][0] +=  two_reg.score(test_df.drop(['Net Income'], axis=1).values, test_df['Net Income'].values)
    results_dict["Polynimal Deg 3 Regression"][0] +=  three_reg.score(test_df.drop(['Net Income'], axis=1).values, test_df['Net Income'].values)
    results_dict["Polynimal Deg 4 Regression"][0] +=  four_reg.score(test_df.drop(['Net Income'], axis=1).values, test_df['Net Income'].values)
    results_dict["Polynimal Deg 5 Regression"][0] +=  five_reg.score(test_df.drop(['Net Income'], axis=1).values, test_df['Net Income'].values)

    results_dict["Linear Regression"][1] +=  mean_squared_error(lin_reg.predict(test_df.drop(['Net Income'], axis=1).values ),test_df['Net Income'])
    results_dict["Lasso Regression"][1] +=  mean_squared_error(las_reg.predict(test_df.drop(['Net Income'], axis=1).values ),test_df['Net Income'])
    results_dict["Ridge Regression"][1] +=  mean_squared_error(ridge_reg.predict(test_df.drop(['Net Income'], axis=1).values ),test_df['Net Income'])
    results_dict["Polynimal Deg 2 Regression"][1] +=  mean_squared_error(two_reg.predict(test_df.drop(['Net Income'], axis=1).values ),test_df['Net Income'])
    results_dict["Polynimal Deg 3 Regression"][1] +=  mean_squared_error(three_reg.predict(test_df.drop(['Net Income'], axis=1).values ),test_df['Net Income'])
    results_dict["Polynimal Deg 4 Regression"][1] +=  mean_squared_error(four_reg.predict(test_df.drop(['Net Income'], axis=1).values ),test_df['Net Income'])
    results_dict["Polynimal Deg 5 Regression"][1] +=  mean_squared_error(five_reg.predict(test_df.drop(['Net Income'], axis=1).values ),test_df['Net Income'])



    if newset == 0:
        results_dict["Linear Regression"][2] = (np.std(df.drop(['Net Income'], axis=1), 0)*lin_reg.coef_).abs()
        results_dict["Lasso Regression"][2] = (np.std(df.drop(['Net Income'], axis=1), 0)*las_reg.coef_).abs()
        results_dict["Ridge Regression"][2] = (np.std(df.drop(['Net Income'], axis=1), 0)*ridge_reg.coef_).abs()
        
    else:     
        results_dict["Linear Regression"][2] = results_dict["Linear Regression"][2].add((np.std(df.drop(['Net Income'], axis=1), 0)*lin_reg.coef_).abs()) 
        results_dict["Lasso Regression"][2] =  results_dict["Lasso Regression"][2].add((np.std(df.drop(['Net Income'], axis=1), 0)*las_reg.coef_).abs())
        results_dict["Ridge Regression"][2] = results_dict["Ridge Regression"][2].add((np.std(df.drop(['Net Income'], axis=1), 0)*ridge_reg.coef_).abs())
        


results_dict["Linear Regression"][2] = results_dict["Linear Regression"][2].divide(N).nlargest(5).to_dict() 
results_dict["Lasso Regression"][2] = results_dict["Lasso Regression"][2].divide(N).nlargest(5).to_dict() 
results_dict["Ridge Regression"][2] = results_dict["Ridge Regression"][2].divide(N).nlargest(5).to_dict() 



    
results_dict["Linear Regression"][0] /= N -1
